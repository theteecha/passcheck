  <script>

    const inputs = [document.getElementById('q1'), document.getElementById('q2'), document.getElementById('q3')];

    const currentAverage = document.getElementById('currentAverage');

    const neededGrade = document.getElementById('neededGrade');

    const message = document.getElementById('message');


    function getGrades() {

      return inputs.map(input => {

        const value = Number(input.value);

        return input.value === '' || Number.isNaN(value) ? null : value;

      });

    }


    function setMessage(text, className) {

      message.textContent = text;

      message.className = className || '';

    }


    function calculate() {

      const grades = getGrades();


      if (grades.some(g => g === null)) {

        currentAverage.textContent = '—';

        neededGrade.textContent = '—';

        setMessage('Enter all three grades to calculate the result.', '');

        return;

      }


      if (grades.some(g => g < 0 || g > 100)) {

        currentAverage.textContent = '—';

        neededGrade.textContent = '—';

        setMessage('Grades should be between 0 and 100.', 'bad');

        return;

      }


      const sum = grades.reduce((a, b) => a + b, 0);

      const avg = sum / 3;

      const rawNeeded = 240 - sum;

      const minimumNeeded = Math.max(0, Math.ceil(rawNeeded));


      currentAverage.textContent = avg.toFixed(1);

      neededGrade.textContent = minimumNeeded;


      if (minimumNeeded === 0) {

        setMessage('You are already on pace to finish with at least a 60 average.', 'good');

      } else if (minimumNeeded <= 100) {

        setMessage(`You need at least a ${minimumNeeded} in Marking Period 4 to finish the year with a 60 average.`, minimumNeeded >= 80 ? 'warn' : 'good');

      } else {

        setMessage(`You would need a ${minimumNeeded}, which is above 100. Talk with your teacher or counselor about available supports and options.`, 'bad');

      }

    }


    function clearForm() {

      inputs.forEach(input => input.value = '');

      calculate();

      inputs[0].focus();

    }


    inputs.forEach(input => input.addEventListener('input', calculate));

    calculate();
